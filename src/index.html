const single_with_room = FileAttachment("data/single-with-roommates.json").json()
const single_no_room = FileAttachment("data/single-no-roommates.json").json()
const with_child = FileAttachment("data/with-child.json").json()
const data = {
  "single_with_room": single_with_room,
  "single_no_room": single_no_room,
  "with_child": with_child
}
const transformData = (data) => {
  let cumulative = 0;
  return data.map(d => {
    const y1 = cumulative;
    const y2 = y1 + d.value;
    cumulative = y2;
    return { ...d, y1, y2 };
  });
};

const formatUSD = new Intl.NumberFormat("en-US", {
  style: "currency",
  currency: "USD"
});
function plotSpending(data, scenario, width) {
  const scen_data = data[scenario];
  const transformedData = transformData(scen_data);


  return Plot.plot({
    width: width,
    height: width * 0.5, // Maintain aspect ratio
    marginLeft: 60, // give space for the y-axis labels
    y: {
      grid: true,
      label: "Dollar Amount ↓",
      nice: true
    },
    x: {
      domain: transformedData.map(d => d.category),
      label: "Categories of Spending →"
    },
    marks: [
      // A horizontal rule at y=0 for reference
      Plot.ruleY([0], { stroke: "black" }),

      // Rectangles from y1 to y2
      Plot.rectY(transformedData, {
        x: "category",
        y1: "y1",
        y2: "y2",
        fill: d => d.value >= 0 ? "green" : "red"
      }),

      // Optional text labels for the value on each bar
      Plot.text(transformedData, {
        x: "category",
        y: d => (d.y1 + d.y2) / 2,
        dy: 5,
        text: d => d.value !== 0 ? formatUSD.format(d.value) : "",
        fill: "white",
        textAnchor: "middle"
      })
    ]
  });
}
